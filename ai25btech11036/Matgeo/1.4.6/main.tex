\item If the point $P(2,1)$ lies on the line segment joining points $A(4,2)$ and $B(8,4)$, then
\begin{enumerate}
    \item $AP = \tfrac{1}{4}AB$
    \item $AP = PE$
    \item $PB = \tfrac{1}{3}AB$
    \item $AP = \tfrac{3}{5}AB$
\end{enumerate}
\textbf{Solution:}\\

% Given:  A(4,2), B(8,4), P(2,1)
\begin{align}
\vec{A} &= \myvec{4\\2},\qquad
\vec{B} = \myvec{8\\4},\qquad
\vec{P} = \myvec{2\\1}.
\end{align}

\begin{align}
\vec{AB} &= \vec{B}-\vec{A}
= \myvec{8\\4}-\myvec{4\\2}
= \myvec{4\\2},\\[2mm]
\vec{AP} &= \vec{P}-\vec{A}
= \myvec{2\\1}-\myvec{4\\2}
= \myvec{-2\\-1}.
\end{align}

\begin{align}
\|\vec{AB}\|^2 &= 4^2 + 2^2 = 20,\\[1mm]
\|\vec{AP}\|^2 &= (-2)^2 + (-1)^2 = 5.
\end{align}

\begin{align}
\left(\frac{AP}{AB}\right)^2
= \frac{\|\vec{AP}\|^2}{\|\vec{AB}\|^2}
= \frac{5}{20}
= \frac{1}{4}
\;\Rightarrow\;
\frac{AP}{AB} = \frac{1}{2}.
\end{align}

\[
\boxed{\,AP=\tfrac{1}{2}\,AB\,}
\]
