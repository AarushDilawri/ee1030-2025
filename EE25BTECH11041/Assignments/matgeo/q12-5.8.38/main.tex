\begin{align}
    y-7=7\times(x-7)\And y+3=3\times(x+3)\\
    7x-y=42\And3x-y=-6
\end{align}

\begin{align}
\begin{pmatrix}7&-1\\3&-1\end{pmatrix}\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}42\\-6\end{pmatrix}\\
\vec{A}\vec{x}=\vec{c}
